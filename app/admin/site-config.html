// Authentication Functions - Use shared Auth module document.addEventListener('DOMContentLoaded', function() { // Use Auth module's isAuthenticated method if (Auth.isAuthenticated()) { document.getElementById('login-section').style.display = 'none'; document.getElementById('dashboard-section').style.display = 'block'; loadSiteConfiguration(); } else { document.getElementById('login-section').style.display = 'block'; document.getElementById('dashboard-section').style.display = 'none'; } // Set up logout event listener const logoutBtn = document.getElementById('logout-btn'); if (logoutBtn) { logoutBtn.addEventListener('click', Auth.logout.bind(Auth)); } });
